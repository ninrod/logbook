* inverting glob patterns in bash

#+BEGIN_SRC sh

#!/usr/bin/env bash
# turns on bash extended glob
shopt -s extglob

# inverts *.cp glob
for file in dot/*.cp; do
  echo $file
done

# turns off bash extended glob
shopt -u extglob
#+END_SRC
